  | { type: "DATA_LOADED"; payload: CalculatorDataV3 }

  | { type: "RESET_ALL" }

export const initialWizardState: WizardState = { /* sane defaults; mirror lines 491–521 of current wizard exactly */ };

export function wizardReducer(state: WizardState, action: WizardAction): WizardState {
  switch (action.type) {
    case "SELECT_PROFILE":
      return { ...state, profileId: action.profileId, modeId: action.modeId, quantity: action.quantity,
               equipmentValueUsd: null, destinationCountry: "", destinationPortKey: null,

import dynamic from "next/dynamic";

// .then(m => ({ default: m.RouteGlobe })) wraps the named export as the

export const RouteGlobeV3 = dynamic(
  () =>
    import("@/components/freight-calculator/route-globe").then((m) => ({
      default: m.RouteGlobe,
    })),
  {
    ssr: false,
    loading: () => (
      <div
        className="relative h-72 w-full animate-pulse overflow-hidden rounded-xl bg-black/80"
        aria-hidden="true"
      />
    ),
  }
);
